# FreshTrack — Stack Migration Plan

**From:** Next.js 14 + Supabase + localStorage
**To:** TanStack Start + Cloudflare Workers + D1 + better-auth + Drizzle ORM

---

## Architecture Overview

```
┌───────────────────────────────────────────────────────┐
│                   Cloudflare Edge                      │
│                                                       │
│  ┌─────────────┐  ┌──────────────┐  ┌──────────────┐ │
│  │  TanStack    │  │  better-auth │  │  Anthropic   │ │
│  │  Start (SSR) │──│  (sessions)  │  │  Haiku API   │ │
│  │  + Router    │  │  + Drizzle   │  │  (recipes)   │ │
│  └──────┬───────┘  └──────┬───────┘  └──────────────┘ │
│         │                 │                            │
│         └────────┬────────┘                            │
│                  ▼                                     │
│         ┌──────────────┐                               │
│         │  Cloudflare  │                               │
│         │  D1 (SQLite) │                               │
│         └──────────────┘                               │
└───────────────────────────────────────────────────────┘
```

**Key principle:** The React UI (components, pages, animations, Tailwind) transfers almost unchanged. The migration is primarily about the **data layer, routing, and auth**.

```ts
  createdAt: text("created_at").notNull().$defaultFn(() => new Date().toISOString()),
  updatedAt: text("updated_at").notNull().$defaultFn(() => new Date().toISOString()),
}, (table) => ({
  householdIdx: index("idx_fi_household").on(table.householdId),
  expiryIdx: index("idx_fi_expiry").on(table.expiryDate),
  locationIdx: index("idx_fi_location").on(table.location),
  barcodeIdx: index("idx_fi_barcode").on(table.barcode),
}));

export const wasteLogs = sqliteTable("waste_logs", {
  id: text("id").primaryKey().$defaultFn(() => crypto.randomUUID()),
  householdId: text("household_id").notNull().references(() => households.id, { onDelete: "cascade" }),
  itemName: text("item_name").notNull(),
  category: text("category", {
    enum: ["dairy","meat","poultry","seafood","produce","grains","beverages","condiments","leftovers","frozen_meals","snacks","other"],
  }).notNull(),
  quantity: real("quantity").notNull().default(1),
  unit: text("unit").notNull().default("item"),
  reason: text("reason", { enum: ["expired","spoiled","leftover","other"] }).notNull(),
  estimatedCost: real("estimated_cost").notNull().default(0),
  wastedDate: text("wasted_date").notNull(),
  createdAt: text("created_at").notNull().$defaultFn(() => new Date().toISOString()),
}, (table) => ({
  householdIdx: index("idx_wl_household").on(table.householdId),
  dateIdx: index("idx_wl_date").on(table.wastedDate),
}));

export const notificationPreferences = sqliteTable("notification_preferences", {
  id: text("id").primaryKey().$defaultFn(() => crypto.randomUUID()),
  userId: text("user_id").notNull().references(() => users.id, { onDelete: "cascade" }).unique(),
  daysBefore: integer("days_before").notNull().default(2),
  enabled: integer("enabled", { mode: "boolean" }).default(true),
  createdAt: text("created_at").notNull().$defaultFn(() => new Date().toISOString()),
});
```

**Key changes from Postgres to D1/SQLite:**
- `uuid` → `text` with `$defaultFn(() => crypto.randomUUID())`
- `timestamptz` → `text` (ISO strings) or `integer` (timestamps)
- `numeric` → `real`
- `boolean` → `integer({ mode: "boolean" })`
- No PostgreSQL extensions, triggers, or RLS — access control moves to application layer (server functions)
- No `plpgsql` — `updated_at` handled in application code

## Phase 2 — Authentication: better-auth

### 2.1 Auth Configuration — `src/auth/index.ts`

**Critical: `getAuth()` must be called PER REQUEST** because D1 bindings are only available inside request handlers.

```ts
import { betterAuth } from "better-auth";
import { drizzleAdapter } from "better-auth/adapters/drizzle";
import { getDb } from "@/db";

export function getAuth(env: Env) {
  const db = getDb(env.DB);

  return betterAuth({
    database: drizzleAdapter(db, { provider: "sqlite" }),
    secret: env.BETTER_AUTH_SECRET,
    baseURL: env.BETTER_AUTH_URL,

    emailAndPassword: { enabled: true },

    session: {
      cookieCache: { enabled: true, maxAge: 60 * 5 }, // 5 min cache
    },

    user: {
      additionalFields: {
        householdId: { type: "string", required: false },
      },
    },
  });
}

export type Auth = ReturnType<typeof getAuth>;
```

```ts
import { getCloudflareEnv } from "@/lib/env";

export const authMiddleware = createMiddleware().server(async ({ next }) => {
  const env = getCloudflareEnv();
  const auth = getAuth(env);

  const session = await auth.api.getSession({ headers: new Headers() });

  if (!session) {
    throw new Error("Unauthorized");
  }

  return next({ context: { session, userId: session.user.id, env } });
});
```

## Phase 3 — Server Functions (Data Layer)

Replace `useFoodItems` localStorage hook with TanStack Query + server functions.

### 3.1 Server Functions — `src/server/food-items.ts`

```ts
import { createServerFn } from "@tanstack/react-start";
import { z } from "zod";
import { getDb, schema } from "@/db";
import { eq, and, asc } from "drizzle-orm";
import { authMiddleware } from "@/middleware/auth";

// ── GET all food items for the user's household ──
export const getFoodItems = createServerFn({ method: "GET" })
  .middleware([authMiddleware])
  .handler(async ({ context }) => {
    const db = getDb(context.env.DB);
    const householdId = await getUserHouseholdId(db, context.userId);
    if (!householdId) return [];

    return db
      .select()
      .from(schema.foodItems)
      .where(eq(schema.foodItems.householdId, householdId))
      .orderBy(asc(schema.foodItems.expiryDate));
  });

// ── ADD a food item ──
const addItemSchema = z.object({
  name: z.string().min(1),
  category: z.enum(["dairy","meat","poultry","seafood","produce","grains","beverages","condiments","leftovers","frozen_meals","snacks","other"]),
  location: z.enum(["fridge","freezer","pantry"]),
  quantity: z.number().positive(),
  unit: z.string(),
  addedDate: z.string(),
  expiryDate: z.string(),
  notes: z.string().nullable(),
  barcode: z.string().nullable(),
  shelf: z.string().nullable(),
});

export const addFoodItem = createServerFn({ method: "POST" })
  .middleware([authMiddleware])
  .validator(addItemSchema)
  .handler(async ({ context, data }) => {
    const db = getDb(context.env.DB);
    const householdId = await getUserHouseholdId(db, context.userId);
    if (!householdId) throw new Error("No household");

    const [item] = await db.insert(schema.foodItems).values({
      ...data,
      householdId,
      opened: false,
      createdBy: context.userId,
    }).returning();

    return item;
  });

// ── UPDATE a food item ──
export const updateFoodItem = createServerFn({ method: "POST" })
  .middleware([authMiddleware])
  .validator(z.object({ id: z.string(), updates: z.record(z.any()) }))
  .handler(async ({ context, data }) => {
    const db = getDb(context.env.DB);
    await db.update(schema.foodItems)
      .set({ ...data.updates, updatedAt: new Date().toISOString() })
      .where(eq(schema.foodItems.id, data.id));
  });

// ── DELETE a food item ──
export const deleteFoodItem = createServerFn({ method: "POST" })
  .middleware([authMiddleware])
  .validator(z.object({ id: z.string() }))
  .handler(async ({ context, data }) => {
    const db = getDb(context.env.DB);
    await db.delete(schema.foodItems).where(eq(schema.foodItems.id, data.id));
  });

// ── MARK as wasted (delete item + insert waste log) ──
export const markWasted = createServerFn({ method: "POST" })
  .middleware([authMiddleware])
  .validator(z.object({
    id: z.string(),
    reason: z.enum(["expired", "spoiled", "leftover", "other"]),
  }))
  .handler(async ({ context, data }) => {
    const db = getDb(context.env.DB);
    const [item] = await db.select().from(schema.foodItems)
      .where(eq(schema.foodItems.id, data.id));
    if (!item) throw new Error("Item not found");

    // Insert waste log
    await db.insert(schema.wasteLogs).values({
      householdId: item.householdId,
      itemName: item.name,
      category: item.category,
      quantity: item.quantity,
      unit: item.unit,
      reason: data.reason,
      estimatedCost: estimateItemCost(item.category, item.quantity),
      wastedDate: new Date().toISOString().split("T")[0],
    });

    // Delete the food item
    await db.delete(schema.foodItems).where(eq(schema.foodItems.id, data.id));
  });

// ── Waste logs ──
export const getWasteLogs = createServerFn({ method: "GET" })
  .middleware([authMiddleware])
  .handler(async ({ context }) => {
    const db = getDb(context.env.DB);
    const householdId = await getUserHouseholdId(db, context.userId);
    if (!householdId) return [];

    return db.select().from(schema.wasteLogs)
      .where(eq(schema.wasteLogs.householdId, householdId))
      .orderBy(asc(schema.wasteLogs.wastedDate));
  });

// ── Helper: get user's household ──
async function getUserHouseholdId(db: any, userId: string): Promise<string | null> {
  const [membership] = await db.select()
    .from(schema.householdMembers)
    .where(eq(schema.householdMembers.userId, userId))
    .limit(1);
  return membership?.householdId ?? null;
}

// ── Helper: cost estimation (moved from client) ──
function estimateItemCost(category: string, quantity: number): number {
  const costMap: Record<string, number> = {
    dairy: 3.5, meat: 8, poultry: 6, seafood: 10, produce: 2.5,
    grains: 3, beverages: 2, condiments: 4, leftovers: 5,
    frozen_meals: 5, snacks: 3.5, other: 3,
  };
  return (costMap[category] || 3) * quantity;
}
```

```ts
import { createServerFn } from "@tanstack/react-start";
import { z } from "zod";
import { getDb, schema } from "@/db";
import { eq } from "drizzle-orm";
import { authMiddleware } from "@/middleware/auth";

export const createHousehold = createServerFn({ method: "POST" })
  .middleware([authMiddleware])
  .validator(z.object({ name: z.string().min(1) }))
  .handler(async ({ context, data }) => {
    const db = getDb(context.env.DB);
    const inviteCode = crypto.randomUUID().slice(0, 8);

    const [household] = await db.insert(schema.households).values({
      name: data.name,
      createdBy: context.userId,
      inviteCode,
    }).returning();

    // Add creator as owner
    await db.insert(schema.householdMembers).values({
      householdId: household.id,
      userId: context.userId,
      role: "owner",
    });

    return household;
  });

  .validator(z.object({ inviteCode: z.string().min(1) }))
  .handler(async ({ context, data }) => {
    const db = getDb(context.env.DB);
    const [household] = await db.select().from(schema.households)
      .where(eq(schema.households.inviteCode, data.inviteCode));
    if (!household) throw new Error("Invalid invite code");

    await db.insert(schema.householdMembers).values({
      householdId: household.id,
      userId: context.userId,
      role: "member",
    });

    return household;
  });
```

### 4.2 Replace `useFoodItems` — `src/hooks/useFoodItems.ts`

Rewrite the hook using TanStack Query + server functions:

```ts
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { getFoodItems, addFoodItem, updateFoodItem, deleteFoodItem, markWasted } from "@/server/food-items";
import { getWasteLogs } from "@/server/food-items";
import { getExpiryStatus, sortByExpiry } from "@/lib/utils";
import { useMemo } from "react";

export function useFoodItems() {
  const qc = useQueryClient();

  const { data: items = [], isLoading } = useQuery({
    queryKey: ["food-items"],
    queryFn: () => getFoodItems(),
  });

  const { data: wasteLogs = [] } = useQuery({
    queryKey: ["waste-logs"],
    queryFn: () => getWasteLogs(),
  });

  const addItemMut = useMutation({
    mutationFn: addFoodItem,
    onSuccess: () => qc.invalidateQueries({ queryKey: ["food-items"] }),
  });

  const updateItemMut = useMutation({
    mutationFn: (args: { id: string; updates: any }) => updateFoodItem(args),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["food-items"] }),
  });

  const deleteItemMut = useMutation({
    mutationFn: (id: string) => deleteFoodItem({ id }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ["food-items"] }),
  });

  const markWastedMut = useMutation({
    mutationFn: (args: { id: string; reason: string }) => markWasted(args),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ["food-items"] });
      qc.invalidateQueries({ queryKey: ["waste-logs"] });
    },
  });

  // Same computed values as before — this logic is unchanged
  const itemsByLocation = useMemo(() => { /* same */ }, [items]);
  const expiringItems = useMemo(() => { /* same */ }, [items]);
  const expiredItems = useMemo(() => { /* same */ }, [items]);
  const stats = useMemo(() => { /* same */ }, [items, itemsByLocation, expiringItems, expiredItems]);

  return {
    items,
    wasteLogs,
    isLoaded: !isLoading,
    itemsByLocation,
    expiringItems,
    expiredItems,
    stats,
    addItem: addItemMut.mutateAsync,
    updateItem: (id: string, updates: any) => updateItemMut.mutateAsync({ id, updates }),
    deleteItem: deleteItemMut.mutateAsync,
    markAsWasted: (id: string, reason: string) => markWastedMut.mutateAsync({ id, reason }),
    markAsConsumed: deleteItemMut.mutateAsync,
    toggleOpened: (id: string) => {
      const item = items.find((i) => i.id === id);
      if (!item) return;
      updateItemMut.mutate({
        id,
        updates: { opened: !item.opened, openedDate: !item.opened ? new Date().toISOString().split("T")[0] : null },
      });
    },
  };
}
```

**The return shape is identical** to the current hook, so **no page/component code changes are needed**.

## Phase 6 — Auth UI

### 6.1 Update Auth Page

Replace the mock `handleSubmit` with real calls to the better-auth client:

```tsx
import { signIn, signUp } from "@/auth/client";
import { useNavigate } from "@tanstack/react-router";

const handleSubmit = async (e: React.FormEvent) => {
  e.preventDefault();
  if (mode === "login") {
    const { error } = await signIn.email({ email, password });
    if (!error) navigate({ to: "/" });
  } else {
    const { error } = await signUp.email({ email, password, name: email.split("@")[0] });
    if (!error) {
      // Create/join household then navigate
      if (joinMode === "create") await createHousehold({ name: householdName });
      else await joinHousehold({ inviteCode });
      navigate({ to: "/" });
    }
  }
};
```

### 6.2 Protected Routes

Add a `beforeLoad` guard on protected routes:

```tsx
export const Route = createFileRoute("/items")({
  beforeLoad: async () => {
    const session = await getSession();  // server function
    if (!session) throw redirect({ to: "/auth" });
  },
  component: ItemsPage,
});
```
